Architektura Systemu

MoodBook to hybrydowa aplikacja dziennika nastroju, która wykorzystuje zaawansowane modele sztucznej inteligencji oraz bazę wektorową do sprawdzania skuteczności zaleceń i unikania ponownego proponowania nieskutecznych metod.

Kluczowe Mechanizmy

  • 🧠
    Inteligentna Pętla Retry (Weryfikacja Wektorowa):

    Backend po wygenerowaniu porad przez Gemini konwertuje je na wektory (embeddingi) i porównuje z nieskutecznymi poradami użytkownika. Jeśli podobieństwo cosinusowe przekracza próg 0.85, odrzuca poradę i generuje nową.

  • 🔄
    Pętla Feedbacku (Zamykanie Cyklu):

    Przy każdym nowym wpisie system sprawdza, czy dawne problemy (ból fizyczny lub stres w notatkach) ustąpiły. W zależności od tego oznacza poprzednio stosowane porady jako skuteczne lub nieskuteczne.

  • ⚠️
    Obsługa Błędów Rate Limit (429):

    Aplikacja nie podaje sztucznych porad mockowych. W razie przeciążenia darmowego serwera, użytkownik otrzymuje jasny monit o błędzie z opcją natychmiastowego ponowienia zapytania ("Retry").

async function addNote(userId, content, ammount_sleep, ammount_of_water, nutrition_intake, date_added) {
    try {
        // Określenie daty dodania wpisu (domyślnie aktualny czas)
        const dateToUse = date_added ? new Date(date_added) : new Date();
        // Wykonanie sparametryzowanego zapytania SQL INSERT

            `INSERT INTO Notes (UserId, Content, Ammout_sleep, Ammout_of_water, Nutrition_intake, Date_added)

2. Analiza AI & Wektorowa

Jak sztuczna inteligencja wykrywa nastroje i wektorowo (za pomocą podobieństwa cosinusowego) odrzuca porady podobne do nieskutecznych.

Serwer sprawdza, które porady w przeszłości nie zadziałały u danego użytkownika i pobiera ich wektory (embeddingi).

Jeśli wygenerowana rada ma podobieństwo do złej rady wyższe niż 0.85, backend odrzuca tę próbę i prosi AI o nową.

async function analyzeNoteHandler(req, res) {
  try {
    // Pobranie danych wpisu z ciała żądania HTTP
    const { noteData, noteId, userId } = req.body;
    if (!noteData) return res.status(400).json({ error: 'Brak noteData' });

    // Domyślne ustawienie ID użytkownika
    const uid = userId || noteData.userId || 1;

    // Pobranie z bazy danych wszystkich nieskutecznych i skutecznych porad tego użytkownika (potrzebnych do promptu i weryfikacji wektorowej)
    const [ineffective, effective] = await Promise.all([
      getIneffectiveAdvice(uid),
      getEffectiveAdvice(uid),
    ]);

    // Wywołanie głównej usługi analizy AI z weryfikacją kolizji wektorowych
    const analysis = await analyzeNote(noteData, ineffective, effective);

    let savedAdvices = [];
    if (noteId) {
      const advices = [analysis.porada1, analysis.porada2, analysis.porada3].filter(Boolean);
      // Równolegle: 
      // 1. Zapisujemy nowo wygenerowane porady w tabeli AiAdvice
      // 2. Aktualizujemy emocjonalną analizę (szczęście, stres itp.) bezpośrednio w tabeli Notes
      // 3. Generujemy i zapisujemy embedding wektorowy całej treści notatki
      [savedAdvices] = await Promise.all([
        saveAdvice(uid, noteId, advices),
        updateNoteAnalysis(noteId, {
          nastroj: analysis.nastroj,
          szczescie: analysis.szczescie,
          smutek: analysis.smutek,
          stres: analysis.stres,
          zlosc: analysis.zlosc,
        }),
        generateEmbedding(noteData.content || '').then(emb => updateNoteEmbedding(noteId, emb))
      ]);
    }

    // Zwrócenie sformatowanej analizy oraz identyfikatorów zapisanych porad do frontendu
    res.status(200).json({ analysis, adviceIds: savedAdvices.map(a => a.id) });
  } catch (err) {
    console.error('Błąd analizy AI:', err);
    res.status(500).json({ error: 'Błąd analizy AI', details: err.message });

  let currentPrompt = prompt;
  let resultAnalysis = null;

  // Główna pętla retry (maksymalnie 3 podejścia w przypadku kolizji wektorowych)

    } catch (err) {
      console.error("Błąd API OpenAI (analiza wpisu):", err.message);
      throw err;
    }

    let collisionDetected = false;
    let collisionText = "";

    const newAdvices = [resultAnalysis.porada1, resultAnalysis.porada2, resultAnalysis.porada3].filter(Boolean);

    // Weryfikacja wektorowa każdej wygenerowanej porady z osobna
    for (const adv of newAdvices) {
      if (!adv.text) continue;
      
      let newEmb = null;
      try {
        // Wygenerowanie wektora (embeddingu) dla wygenerowanej przez AI porady
        newEmb = await generateEmbedding(adv.text);
      } catch (e) {
        console.warn("Błąd generowania embeddingu podczas weryfikacji:", e.message);
      }

      if (newEmb) {
        // Porównanie z bazą nieskutecznych embeddingów użytkownika
        for (const bad of ineffectiveAdvice) {
          if (bad.embedding) {
            // Obliczenie podobieństwa cosinusowego wektorów
            const sim = cosineSimilarity(newEmb, bad.embedding);
            // Jeśli podobieństwo przekracza próg 0.85 - mamy kolizję semantyczną
            if (sim > 0.85) {
              collisionDetected = true;
              collisionText = bad.advicetext;
              console.log(`[AI Verification] Kolizja! Nowa porada "${adv.text}" jest zbyt podobna do nieskutecznej porady: "${bad.advicetext}" (Podobieństwo: ${sim.toFixed(2)})`);
              break;
            }
          }
        }
      }
      if (collisionDetected) break;
    }

    // Jeśli brak kolizji - wychodzimy z pętli i zwracamy poprawny zestaw porad
    if (!collisionDetected) {
      return resultAnalysis;
    }

    // W przypadku kolizji modyfikujemy prompt dla kolejnej próby (dopisując instrukcję negatywną)
    currentPrompt += `\n\n[System] UWAGA: W tej próbie wygenerowałeś poradę zbyt podobną do nieskutecznej porady użytkownika: "${collisionText}". Spróbuj wymyślić inną strategię i sformułowanie, omijając to podobieństwo!`;
    attempt++;
  }

  return resultAnalysis;
}

function cosineSimilarity(vecA, vecB) {
  // Sprawdzenie, czy wektory istnieją i posiadają taką samą długość (np. 1024 wymiary dla modelu stella)
  if (!vecA || !vecB || vecA.length !== vecB.length) return 0;
  let dotProduct = 0.0;
  let normA = 0.0;
  let normB = 0.0;
  
  // Obliczenie iloczynu skalarnego oraz norm euklidesowych w pętli
  for (let i = 0; i < vecA.length; i++) {
    dotProduct += vecA[i] * vecB[i];
    normA += vecA[i] * vecA[i];
    normB += vecB[i] * vecB[i];
  }
  
  // Zabezpieczenie przed dzieleniem przez zero
  if (normA === 0 || normB === 0) return 0;
  
  // Zwrócenie podobieństwa cosinusowego (wynik z przedziału [-1, 1])
  return dotProduct / (Math.sqrt(normA) * Math.sqrt(normB));
}

  // Iteracja po wszystkich wygenerowanych przez model poradach (1-3)
  for (const adv of advices) {
    let embedding = null;
    try {
      // Wygenerowanie wektora osadzenia (embeddingu) dla tekstu porady
      embedding = await generateEmbedding(adv.text || '');
    } catch (err) {
      console.error('Błąd generowania embeddingu dla porady:', err);
    }
    // Zapis porady w tabeli AiAdvice w PostgreSQL
    const res = await pool.query(
      `INSERT INTO AiAdvice (UserId, NoteId, AdviceText, Category, Embedding)
       VALUES ($1, $2, $3, $4, $5) RETURNING *`,
      [userId, noteId, adv.text, adv.category || 'ogólne', embedding]
    );
    saved.push(res.rows[0]);
  }
  return saved;
}

async function markAdviceFeedback(adviceId, wasFollowed, issuePersists) {
  let wasEffective = null;
  // Wyznaczenie skuteczności porady:
  // Była skuteczna (WasEffective = true) tylko wtedy, gdy pacjent z niej skorzystał (wasFollowed = true) 
  // i problem nie występuje kolejnego dnia (issuePersists = false)
  if (wasFollowed !== null && issuePersists !== null) {
    wasEffective = wasFollowed && !issuePersists;
  }

  // Zaktualizowanie statusu w tabeli AiAdvice w bazie danych PostgreSQL
  const res = await pool.query(
    `UPDATE AiAdvice SET WasFollowed = $1, WasEffective = $2
     WHERE Id = $3 RETURNING *`,
    [wasFollowed, wasEffective, adviceId]
  );
  return res.rows[0];
}
